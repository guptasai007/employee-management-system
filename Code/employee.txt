    private void applyleaveActionPerformed(java.awt.event.ActionEvent evt) {                                           
     try{
      pst=con.prepareStatement("update employees set subject=? where empid=?");
      pst.setString(1,leavesub.getText());
      pst.setInt(2,id);
      pst.executeUpdate();
      pst=con.prepareStatement("update employees set fromdate=? where empid=?");
      pst.setString(1,formatter2.format(parser2.parse(leavefrom.getText())));
      pst.setInt(2,id);
      pst.executeUpdate();
      pst=con.prepareStatement("update employees set todate=? where empid=?");
      pst.setString(1,formatter2.format(parser2.parse(leaveto.getText())));
      pst.setInt(2,id);
      pst.executeUpdate();
      pst=con.prepareStatement("update employees set leaverequest=? where empid=?");
      pst.setString(1,"y");
      pst.setInt(2,id);
      pst.executeUpdate();
      leaveprompt.setText("Applied for Leave for above said Dates!");
     }
     catch(SQLException s){s.printStackTrace();}
     catch(ParseException p){p.printStackTrace(); leaveprompt.setText("Please Provide Dates in proper Format!");}
     applyleave.setEnabled(false);
    }
